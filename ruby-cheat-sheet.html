Ruby Cheat Sheet

Variables

x = "string" 
	=>"string"
x = 6 
	=> 6

Methods

def my_method(x,y)
	x + y
end
	
my_method(2,4)
	=> 6

Loops

For

for i in (0..5) do
    p i
end
	=> 1
	   2
	   3
	   4
	   5

While

i = 0
while i < 5 do
    p i
    i +=1
end
	=> 1
	   2
	   3
	   4

Hashes

hash = {'one' => 1, 'two' => 2, 'three' => 3}
hash['two']
	=> 2
hash_symbol_key = {one: 1, two: 2, three: 3}
hash[:one]
	=> 1
hash_symbol_key[:four] = 4
p hash_symbol_key
	=> {one: 1, two: 2, three: 3, four: 4}

Conditionals

x = 8
y = 3

if x < y
    puts "#{x} is less than #{y}"
else
    p "#{x} is greater than #{y}"
end
		=> "8 is greater than 3"

Arrays

array = ['a','b','c','d','e']

array[2]
	=> 'c' 

array.push('f')
	=> ['a','b','c','d','e','f']

array.pop
	=> 'f'

array.length
	=> 5 #after we popped 'f'
