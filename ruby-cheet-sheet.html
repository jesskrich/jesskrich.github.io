y = 6

    => 10
y - z
    => 2
    
#multiplication/division
x * z 
    => 16

    => 1 #ruby rounds down to the nearest
        #whole number when dividing integers
        #if there is a remainder

Conditionals

x = 8
y = 3

if x < y
    puts "#{x} is less than #{y}"
else
    p "#{x} is greater than #{y}"
end
		=> "8 is greater than 3"

Arrays

array[2]
	=> 'c' 

array.push('f')
	=> ['a','b','c','d','e','f']

array.pop
	=> 'f'

array.length
	=> 5 #after we popped 'f'
